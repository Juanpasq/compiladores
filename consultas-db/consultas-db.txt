Por Frente, obtener la cantidad de personal asignado y su estado.

db.empleados_talles.aggregate([
  {
    $group: {

      // Agrupamos por dos campos a la vez
      _id: {
        frente_trabajo: "$frente_trabajo",
        estado: "$estado"
      },
      // Contamos cuantos documentos caen en este grupo
      cantidad_empleados: { $sum: 1 }
    }
  },
  {
    
    $sort: { "_id.frente_trabajo": 1, "_id.estado": 1 }
  }
])

Obtener la distribución de Sector/Nomenclador por Categoría.

db.empleados_talles.aggregate([

  //contamos cuántas personas hay por cada combinación exacta
  {
    $group: {
      _id: {
        categoria: "$categoria",
        nomenclador: "$nomenclador"
      },
      cantidad: { $sum: 1 }
    }
  },
  // re-agrupamos  por categoría, y armamos la distribución en una lista
  {
    $group: {
      _id: "$_id.categoria", // El grupo principal ahora es la Categoría
      distribucion_sectores: {
        $push: {
          nomenclador: "$_id.nomenclador",
          cantidad: "$cantidad"
        }
      },
      total_en_categoria: { $sum: "$cantidad" } // ¡Un dato extra que le suma al TP!
    }
  },
  
  {
    $sort: { "_id": 1 }
  }
])

Obtener la distribución de tallas por prenda/accesorios sobre:
 
- Personal activo/inactivo.

db.empleados_talles.aggregate([
  // Transformar el objeto de talles en un Array que Mongo pueda leer
  // esto convierte { pantalon: "48" } en [{ k: "pantalon", v: "48" }]
  {
    $project: {
      estado: 1, // Mantenemos el campo Estado
      talles_array: { $objectToArray: "$talles" } 
    }
  },
  
  // separar ese array para que cada prenda tenga su propio registro temporal
  {
    $unwind: "$talles_array"
  },
  
  // agrupamos y contamos cada combinacion exacta (estado + prenda + talle)
  {
    $group: {
      _id: {
        estado: "$estado",
        prenda: "$talles_array.k", // "k" es la key (ej: pantalon)
        talle: "$talles_array.v"   // "v" es el value (ej: "48")
      },
      cantidad: { $sum: 1 }
    }
  },
  
  // ordenamos para la distribucion
  {
    $group: {
      _id: {
        estado: "$_id.estado",
        prenda: "$_id.prenda"
      },
      distribucion_talles: {
        $push: {
          talle: "$_id.talle",
          cantidad: "$cantidad"
        }
      },
      total_unidades: { $sum: "$cantidad" }
    }
  },
  
  // alfabeticamente
  {
    $sort: { "_id.estado": 1, "_id.prenda": 1 }
  }
])
